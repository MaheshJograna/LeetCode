# 1.Two Sum
Problem: https://leetcode.com/problems/two-sum/description/
```c
Solution:
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<int> num;
        for (int i = 0; i != nums.size(); i++)
        {
            int sum;
            for(int j = 0; j !=  nums.size(); j++)
            {
                sum = nums[i] + nums[j];
                if (sum == target){
                num = {i,j};
                break;

                }
                       
            }
        }

     return num;   
    }

};
